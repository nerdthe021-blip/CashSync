function doRegister(){
  const name=document.getElementById('reName').value.trim();
  const email=document.getElementById('reEmail').value.trim().toLowerCase();
  const pw=document.getElementById('rePw').value;
  const pw2=document.getElementById('rePw2').value;
  const err=document.getElementById('reErr');
  err.textContent='';

  if(!name) return err.textContent='Please enter your name.';
  if(!email||!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return err.textContent='Please enter a valid email.';
  if(pw.length<6) return err.textContent='Password must be at least 6 characters.';
  if(pw!==pw2) return err.textContent='Passwords do not match.';

  const users=getUsers();
  if(users[email]) return err.textContent='An account with this email already exists.';

  // Simple hash (not cryptographic — for demo purposes)
  users[email]={name,pwHash:simpleHash(pw)};
  saveUsers(users);

  // Auto-login
  currentUser={email,name};
  saveSession(currentUser);
  startApp();
  toast('Welcome, '+name+'! 🎉','ok');
}

function doLogin(){
  const email=document.getElementById('liEmail').value.trim().toLowerCase();
  const pw=document.getElementById('liPw').value;
  const err=document.getElementById('liErr');
  err.textContent='';

  if(!email) return err.textContent='Please enter your email.';
  if(!pw) return err.textContent='Please enter your password.';

  const users=getUsers();
  const user=users[email];
  if(!user||user.pwHash!==simpleHash(pw)) return err.textContent='Invalid email or password.';

  currentUser={email,name:user.name};
  saveSession(currentUser);
  startApp();
  toast('Welcome back, '+user.name+'! 👋','ok');
}

// Very simple hash for demo — NOT secure for production
function simpleHash(str){
  let h=0;
  for(let i=0;i<str.length;i++){ h=((h<<5)-h)+str.charCodeAt(i); h|=0; }
  return h.toString(36);
}

function fmt(n){ return (S.currency||'₹')+Math.abs(n).toLocaleString('en-IN',{minimumFractionDigits:2,maximumFractionDigits:2}); }
function fmtShort(n){
  if(n>=1e7) return (S.currency||'₹')+(n/1e7).toFixed(1)+'Cr';
  if(n>=1e5) return (S.currency||'₹')+(n/1e5).toFixed(1)+'L';
  if(n>=1e3) return (S.currency||'₹')+(n/1e3).toFixed(1)+'K';
  return fmt(n);
}

function renderDash(){
  const inc=S.txns.filter(t=>t.a>0).reduce((s,t)=>s+t.a,0);
  const exp=S.txns.filter(t=>t.a<0).reduce((s,t)=>s+Math.abs(t.a),0);
  const bal=inc-exp;
  document.getElementById('s-bal').textContent=(bal<0?'-':'')+fmt(bal);
  document.getElementById('s-inc').textContent=fmt(inc);
  document.getElementById('s-exp').textContent=fmt(exp);
  const rec=[...S.txns].sort((a,b)=>new Date(b.d)-new Date(a.d)).slice(0,5);
  const el=document.getElementById('recentList');
  el.innerHTML=rec.length ? rec.map(t=>txHTML(t,false)).join('') : emptyHTML('💸','No transactions yet','Add your first one!');
  renderPie('dashPie','dashLegend','dash');
  renderDashBudget();
  renderDashTrend();
}

function renderDashBudget(){
  const card=document.getElementById('dashBudgetCard');
  if(!S.budget||S.budget<=0){
    card.innerHTML=`<div class="flex-bet"><span class="txt-m">No budget goal set.</span><button class="btn btn-g btn-sm" onclick="go('budget')">Set Budget →</button></div>`;
    return;
  }
  const now=new Date(),mo=now.getMonth(),yr=now.getFullYear();
  const spent=S.txns.filter(t=>t.a<0&&new Date(t.d).getMonth()===mo&&new Date(t.d).getFullYear()===yr).reduce((s,t)=>s+Math.abs(t.a),0);
  const pct=Math.min((spent/S.budget)*100,100);
  const over=spent>S.budget;
  const col=pct<60?'var(--green)':pct<90?'var(--orange)':'var(--red)';
  card.innerHTML=`
    <div class="flex-bet" style="margin-bottom:8px;">
      <span style="font-size:14px;font-weight:600;">Monthly Budget</span>
      <button class="btn btn-g btn-sm" onclick="go('budget')">Edit</button>
    </div>
    <div class="flex-bet" style="font-size:12.5px;"><span>${fmt(spent)} spent</span><span style="color:${col};font-weight:700;">${pct.toFixed(1)}% used</span></div>
    <div class="prog-wrap"><div class="prog-fill" style="width:${pct}%;background:${col};"></div></div>
    <div class="prog-labels"><span>Budget: ${fmt(S.budget)}</span><span>${over?'⚠️ Over budget!':fmt(S.budget-spent)+' remaining'}</span></div>
    ${over?`<div class="warning-box">⚠️ Over budget by ${fmt(spent-S.budget)}!</div>`:''}`;
}

function renderDashTrend(){
  const el=document.getElementById('dashTrend');
  const now=new Date(),mo=now.getMonth(),yr=now.getFullYear();
  const cats={};
  S.txns.filter(t=>t.a<0&&new Date(t.d).getMonth()===mo&&new Date(t.d).getFullYear()===yr)
    .forEach(t=>{ const c=t.c||'Other'; cats[c]=(cats[c]||0)+Math.abs(t.a); });
  const entries=Object.entries(cats).sort((a,b)=>b[1]-a[1]).slice(0,5);
  if(!entries.length){ el.innerHTML='<p class="txt-m" style="padding:10px 0;">No expenses this month yet.</p>'; return; }
  const max=entries[0][1];
  el.innerHTML=entries.map(([c,v])=>{
    const pct=(v/max)*100,col=CAT_COLORS[c]||'#adb5bd';
    return `<div class="trend-bar-row"><div class="trend-label">${c}</div>
      <div class="trend-bar-bg"><div class="trend-bar-fill" style="width:${pct}%;background:${col};"></div></div>
      <div class="trend-val" style="color:${col};">${fmtShort(v)}</div></div>`;
  }).join('');
}

function renderPie(cid,lid,prefix){

  const canvas=document.getElementById(cid);
  const legEl=document.getElementById(lid);
  if(!canvas) return;
  if(!labels.length){
    const wrap=canvas.closest('.chart-wrap')||canvas.parentElement;
    wrap.innerHTML=emptyHTML('📊','No expense data yet','');
    if(legEl) legEl.innerHTML=''; return;
  }
  if(prefix==='dash'&&dashPie){ dashPie.destroy(); dashPie=null; }

function renderTxList(){
  const q=(document.getElementById('q')?.value||'').toLowerCase();
  const cat=document.getElementById('fCat')?.value||'';
  const type=document.getElementById('fType')?.value||'';
  const month=document.getElementById('fMonth')?.value||'';
  const sort=document.getElementById('fSort')?.value||'new';
  let txns=[...S.txns];
  if(q) txns=txns.filter(t=>t.desc.toLowerCase().includes(q)||(t.c||'').toLowerCase().includes(q)||(t.note||'').toLowerCase().includes(q));
  if(cat) txns=txns.filter(t=>t.c===cat);
  if(type==='i') txns=txns.filter(t=>t.a>0);
  if(type==='e') txns=txns.filter(t=>t.a<0);
  if(month){ const[yr,mo]=month.split('-').map(Number); txns=txns.filter(t=>{ const d=new Date(t.d); return d.getFullYear()===yr&&d.getMonth()+1===mo; }); }
  if(sort==='new') txns.sort((a,b)=>new Date(b.d)-new Date(a.d));
  if(sort==='old') txns.sort((a,b)=>new Date(a.d)-new Date(b.d));
  if(sort==='hi')  txns.sort((a,b)=>Math.abs(b.a)-Math.abs(a.a));
  if(sort==='lo')  txns.sort((a,b)=>Math.abs(a.a)-Math.abs(b.a));

  if(cnt) cnt.textContent=txns.length ? `${txns.length} transaction${txns.length>1?'s':''}` : '';
  el.innerHTML=txns.length ? txns.map(t=>txHTML(t,true)).join('') : emptyHTML('📋','No transactions found','Try adjusting your filters');
}

// ── FIXED COPY FUNCTION ──
async function copyBackup(){
  const box=document.getElementById('expBox');
  const code=box.textContent.trim();
  const btn=document.getElementById('copyBtn');
  if(!code||code.startsWith('Click')) return toast('Generate a backup first.','er');

  let copied=false;

  // Method 1: Clipboard API (modern browsers)
  if(navigator.clipboard&&window.isSecureContext){
    try{
      await navigator.clipboard.writeText(code);
      copied=true;
    }catch(e){ /* fall through */ }
  }

  // Method 2: execCommand fallback (iframes, older browsers)
  if(!copied){
    try{
      const ta=document.createElement('textarea');
      ta.value=code;
      ta.style.cssText='position:fixed;left:-9999px;top:-9999px;opacity:0;';
      document.body.appendChild(ta);
      ta.focus();
      ta.select();
      ta.setSelectionRange(0,code.length);
      copied=document.execCommand('copy');
      document.body.removeChild(ta);
    }catch(e){ copied=false; }
  }

  if(copied){
    // Visual feedback on button
    btn.textContent='✅ Copied!';
    btn.classList.add('copy-success');
    setTimeout(()=>{ btn.textContent='📋 Copy'; btn.classList.remove('copy-success'); },2000);
    toast('Backup code copied to clipboard!','ok');
  }else{
    // Last resort — select text so user can Ctrl+C
    const range=document.createRange();
    range.selectNodeContents(box);
    const sel=window.getSelection();
    sel.removeAllRanges();
    sel.addRange(range);
    toast('Auto-copy unavailable — text selected, press Ctrl+C / ⌘C.','nf');
  }
}

function dlBackup(){

  const a=Object.assign(document.createElement('a'),{

  });

}

function doImport(){
  const code=document.getElementById('impBox').value.trim();
  if(!code) return toast('Paste your backup code first.','er');
  try{
    const binary=atob(code);
    const bytes=Uint8Array.from(binary,c=>c.charCodeAt(0));
    const parsed=JSON.parse(new TextDecoder().decode(bytes));
    if(!Array.isArray(parsed.txns)) throw new Error('Missing txns array');
    S={...S,...parsed};
    saveUserData(); renderAll(); populateCatFilter(); populateMonthFilter(); applyTheme();
    document.getElementById('impBox').value='';
    toast('Data restored successfully!','ok');
  }catch(e){ toast('Invalid backup code. Please check and try again.','er'); }
}

function applyTheme(){
  if(S.theme==='light'){ document.body.classList.add('light'); document.getElementById('themeIco').textContent='🌙'; document.getElementById('themeLabel').textContent='Dark Mode'; }
  else{ document.body.classList.remove('light'); document.getElementById('themeIco').textContent='☀️'; document.getElementById('themeLabel').textContent='Light Mode'; }
}

function toast(msg,type='nf'){
  document.querySelector('.toast')?.remove();
  const icons={ok:'✅',er:'❌',nf:'ℹ️'};
  const el=document.createElement('div');
  el.className=`toast ${type}`;
  el.innerHTML=`<span>${icons[type]}</span>${msg}`;
  document.body.appendChild(el);
  setTimeout(()=>el.remove(),3200);
}
